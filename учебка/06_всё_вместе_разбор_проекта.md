# Разбор проекта ShoeShop — как всё работает вместе

---

## Архитектура приложения

```
App.xaml.cs          ← точка входа, открывает LoginWin
    ↓
LoginWin             ← авторизация → заполняет CurrentUser
    ↓
CatalogWindow        ← главное окно, список товаров
    ↓          ↓          ↓
AddProdWin   EditProdWin  DeleteProduct
(добавить)   (изменить)   (удалить прямо в каталоге)
```

## Блок 2 — Авторизация (LoginWin)

**Что делает:**
1. Пользователь вводит email + пароль
2. Ищем в БД пользователя с такими данными
3. Если нашли — сохраняем в `CurrentUser`, открываем каталог
4. Если не нашли — сообщение об ошибке
5. Гость — сразу открываем каталог без авторизации

```csharp
private void LoginBtn_Click(object sender, RoutedEventArgs e)
{
    using var db = new ShoeshopContext();

    // Найти пользователя по email + паролю
    var user = db.Users
        .Include(u => u.Role)  // подгрузить роль
        .FirstOrDefault(u => u.Email == EmailBox.Text && u.Passw == PasswordBox.Password);

    if (user == null)
    {
        MessageBox.Show("Неверный логин или пароль!");
        return;
    }

    // Запомнить кто вошёл
    CurrentUser.Id = user.IdUser;
    CurrentUser.FullName = $"{user.LastName} {user.FirstName} {user.MiddleName}";
    CurrentUser.Role = user.Role?.RoleName ?? "Guest";

    new CatalogWindow().Show();
    this.Close();
}

private void GuestBtn_Click(object sender, RoutedEventArgs e)
{
    CurrentUser.Role = "Guest";
    new CatalogWindow().Show();
    this.Close();
}
```

## Блок 3 — Каталог товаров (CatalogWindow)

Самый сложный файл (~237 строк). Разбираем по частям.

### 3.1 Инициализация окна

```csharp
private ShoeshopContext db = new ShoeshopContext();

public CatalogWindow()
{
    InitializeComponent();

    // Показать имя пользователя
    UserNameLabel.Content = CurrentUser.FullName ?? "Гость";

    // Показать/скрыть кнопки по роли
    if (CurrentUser.Role != "Administrator")
    {
        AddBtn.Visibility = Visibility.Collapsed;
        EditBtn.Visibility = Visibility.Collapsed;
        DeleteBtn.Visibility = Visibility.Collapsed;
    }

    // Заполнить фильтр поставщиков
    var suppliers = db.Suppliers.ToList();
    suppliers.Insert(0, new Supplier { IdSup = 0, SupName = "Все поставщики" });
    SupplierFilter.ItemsSource = suppliers;
    SupplierFilter.SelectedIndex = 0;

    // Загрузить товары
    LoadProducts();
}
```

### 3.2 Метод загрузки товаров (главный)

```csharp
private void LoadProducts()
{
    // Загрузить всё из БД
    var products = db.Products
        .Include(p => p.Cat)
        .Include(p => p.Manuf)
        .Include(p => p.Sup)
        .Include(p => p.ProdNameNavigation)
        .ToList();

    // Поиск
    string search = SearchBox.Text.Trim().ToLower();
    if (!string.IsNullOrEmpty(search))
    {
        products = products.Where(p =>
            (p.Descrip != null && p.Descrip.ToLower().Contains(search)) ||
            (p.Cat?.CatName.ToLower().Contains(search) == true) ||
            (p.Manuf?.ManufName.ToLower().Contains(search) == true) ||
            (p.Sup?.SupName.ToLower().Contains(search) == true)
        ).ToList();
    }

    // Фильтр по поставщику
    if (SupplierFilter.SelectedItem is Supplier sup && sup.IdSup != 0)
    {
        products = products.Where(p => p.SupId == sup.IdSup).ToList();
    }

    // Сортировка
    if (SortAscBtn.IsChecked == true)
        products = products.OrderBy(p => p.Count).ToList();
    else if (SortDescBtn.IsChecked == true)
        products = products.OrderByDescending(p => p.Count).ToList();

    // Привязать к ListBox
    ProductList.ItemsSource = products;
}
```

### 3.4 Удаление товара

```csharp
private void DeleteBtn_Click(object sender, RoutedEventArgs e)
{
    if (ProductList.SelectedItem is not Product selected)
    {
        MessageBox.Show("Выберите товар для удаления");
        return;
    }

    // Проверить — есть ли товар в заказах
    bool inOrders = db.Prodorders.Any(po => po.ProdId == selected.IdProd);
    if (inOrders)
    {
        MessageBox.Show("Невозможно удалить: товар присутствует в заказах");
        return;
    }

    var result = MessageBox.Show($"Удалить товар '{selected.Descrip}'?", 
                                  "Подтверждение", 
                                  MessageBoxButton.YesNo);
    if (result != MessageBoxResult.Yes) return;

    db.Products.Remove(selected);
    db.SaveChanges();
    LoadProducts();  // обновить список
}
```

В `AddProdWindow`:
```csharp
public partial class AddProdWindow : Window
{
    private string _selectedImagePath;  // путь к выбранной картинке

    public AddProdWindow()
    {
        InitializeComponent();
        LoadComboBoxes();
    }

    private void LoadComboBoxes()
    {
        using var db = new ShoeshopContext();
        
        CatCombo.ItemsSource = db.Categories.ToList();
        CatCombo.DisplayMemberPath = "CatName";

        ManufCombo.ItemsSource = db.Manufacrurers.ToList();
        ManufCombo.DisplayMemberPath = "ManufName";

        SupCombo.ItemsSource = db.Suppliers.ToList();
        SupCombo.DisplayMemberPath = "SupName";

        ProdNameCombo.ItemsSource = db.Prodnames.ToList();
        ProdNameCombo.DisplayMemberPath = "ProdName";
    }

    // Выбор картинки
    private void SelectImageBtn_Click(object sender, RoutedEventArgs e)
    {
        OpenFileDialog dialog = new OpenFileDialog();
        dialog.Filter = "Image files|*.jpg;*.jpeg;*.png";

        if (dialog.ShowDialog() == true)
        {
            _selectedImagePath = dialog.FileName;
            
            // Показать превью
            ProductImage.Source = new BitmapImage(new Uri(_selectedImagePath));
        }
    }

    private void SaveBtn_Click(object sender, RoutedEventArgs e)
    {
        // Валидация
        if (string.IsNullOrWhiteSpace(DescripBox.Text))
        { MessageBox.Show("Введите описание"); return; }

        if (!decimal.TryParse(PriceBox.Text, out decimal price) || price <= 0)
        { MessageBox.Show("Некорректная цена"); return; }

        if (CatCombo.SelectedItem == null)
        { MessageBox.Show("Выберите категорию"); return; }

        using var db = new ShoeshopContext();

        // Скопировать картинку в папку Images проекта
        string imageName = null;
        if (_selectedImagePath != null)
        {
            imageName = Path.GetFileName(_selectedImagePath);
            string dest = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Images", imageName);
            File.Copy(_selectedImagePath, dest, overwrite: true);
        }

        // Создать и сохранить продукт
        var product = new Product
        {
            Descrip = DescripBox.Text,
            Price = price,
            Sale = int.TryParse(SaleBox.Text, out int sale) ? sale : 0,
            Count = int.TryParse(CountBox.Text, out int count) ? count : 0,
            CatId = (CatCombo.SelectedItem as Category)?.IdCat,
            ManufId = (ManufCombo.SelectedItem as Manufacrurer)?.IdManuf,
            SupId = (SupCombo.SelectedItem as Supplier)?.IdSup,
            ProdName = (ProdNameCombo.SelectedItem as Prodname)?.IdProdName,
            Image = imageName
        };

        db.Products.Add(product);
        db.SaveChanges();

        MessageBox.Show("Товар успешно добавлен!");
        this.Close();
    }
}
```

```csharp
// EditProdWindow.xaml.cs
public partial class EditProdWindow : Window
{
    private Product _product;
    private string _selectedImagePath;

    public EditProdWindow(Product product)
    {
        InitializeComponent();
        _product = product;
        LoadComboBoxes();
        FillFields();  // заполнить поля текущими значениями
    }

    private void FillFields()
    {
        DescripBox.Text = _product.Descrip;
        PriceBox.Text = _product.Price.ToString();
        SaleBox.Text = _product.Sale?.ToString() ?? "0";
        CountBox.Text = _product.Count.ToString();

        // Выбрать нужный элемент в ComboBox
        CatCombo.SelectedItem = CatCombo.Items
            .OfType<Category>()
            .FirstOrDefault(c => c.IdCat == _product.CatId);

        // Показать текущую картинку
        if (_product.Image != null)
        {
            string path = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Images", _product.Image);
            if (File.Exists(path))
                ProductImage.Source = new BitmapImage(new Uri(path));
        }
    }

    private void SaveBtn_Click(object sender, RoutedEventArgs e)
    {
        using var db = new ShoeshopContext();

        // Найти продукт в ЭТОМ контексте
        var productToEdit = db.Products.FirstOrDefault(p => p.IdProd == _product.IdProd);
        if (productToEdit == null) return;

        // Обновить поля
        productToEdit.Descrip = DescripBox.Text;
        productToEdit.Price = decimal.Parse(PriceBox.Text);
        productToEdit.Sale = int.Parse(SaleBox.Text);
        productToEdit.Count = int.Parse(CountBox.Text);
        productToEdit.CatId = (CatCombo.SelectedItem as Category)?.IdCat;

        // Если выбрали новую картинку
        if (_selectedImagePath != null)
        {
            string imageName = Path.GetFileName(_selectedImagePath);
            string dest = Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Images", imageName);
            File.Copy(_selectedImagePath, dest, overwrite: true);
            productToEdit.Image = imageName;
        }

        db.SaveChanges();
        MessageBox.Show("Товар обновлён!");
        this.Close();
    }
}
```

## Частые ошибки

| Ошибка | Причина | Решение |
|--------|---------|---------|
| `NullReferenceException` на навигационном свойстве | Не добавил `.Include(...)` | Добавить `.Include(p => p.Cat)` |
| Картинка не отображается | Неверный путь или нет конвертера | Проверить `ImagePathConverter` |
| ComboBox пустой | Не установил `ItemsSource` | `CatCombo.ItemsSource = db.Categories.ToList()` |
| Данные не сохраняются | Забыл `db.SaveChanges()` | Добавить после изменений |
| Редактирование не работает | Разные контексты БД | Найти объект через `db.Products.FirstOrDefault(p => p.IdProd == id)` |
